# javascrpt 문법

## 변수형은 3가지 이다

    let a = 10;
    const b = 20;
    var c = 30;
    
## 등호식은 다음과 같다

     (a === 10) ? 'a는 10' : 'a 는 10이 아님' ;
     (a === 10) && 'a는 10';   //false는 없을때

## 따옴표는 쌍따옴표와 구분하지 않는다.

    const a = 'hello';
    const b = "hello";

## 리턴문이 여러줄일때는 괄호가 필요하다.

    return <div> hello </div>;
    
    return(
      <div>
        hello
      </div> );
